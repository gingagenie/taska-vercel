// add if missing:
import { Badge } from "@/components/ui/badge";
import { useLocation, Link } from "wouter";

export default function Customers() {
  const { data: list = [], isLoading } = useQuery({
    queryKey: ["/api/customers"],
    queryFn: customersApi.getAll,
  });
  const [q, setQ] = useState("");
  const [open, setOpen] = useState(false);
  const [, navigate] = useLocation();

  const filtered = list.filter((c: any) =>
    (c.name || "").toLowerCase().includes(q.toLowerCase()) ||
    (c.contact_name || "").toLowerCase().includes(q.toLowerCase()) ||
    (c.email || "").toLowerCase().includes(q.toLowerCase())
  );

  return (
    <div className="p-6 space-y-4">
      <div className="flex items-center justify-between gap-2 flex-wrap">
        <h1 className="text-2xl font-bold">Customers</h1>
        <div className="flex gap-2">
          <Input
            placeholder="Search by company, contact, email…"
            value={q}
            onChange={(e) => setQ(e.target.value)}
            className="w-72"
          />
          <Button onClick={() => setOpen(true)}>New Customer</Button>
        </div>
      </div>

      {isLoading ? (
        <Card><CardContent>Loading…</CardContent></Card>
      ) : filtered.length === 0 ? (
        <Card><CardContent>No customers found</CardContent></Card>
      ) : (
        <div className="overflow-x-auto bg-white border border-gray-200 rounded-xl">
          <table className="w-full text-sm">
            <thead className="bg-gray-50 text-gray-600 text-xs uppercase">
              <tr>
                <th className="px-4 py-3 text-left">Company</th>
                <th className="px-4 py-3 text-left">Contact</th>
                <th className="px-4 py-3 text-left">Email</th>
                <th className="px-4 py-3 text-left">Phone</th>
                <th className="px-4 py-3 text-left">Address</th>
                <th className="px-4 py-3 text-left w-32">Actions</th>
              </tr>
            </thead>
            <tbody className="divide-y divide-gray-200">
              {filtered.map((c: any) => {
                const addr = [c.street, c.suburb, c.state, c.postcode].filter(Boolean).join(", ");
                return (
                  <tr
                    key={c.id}
                    className="hover:bg-gray-50 cursor-pointer"
                    onClick={() => navigate(`/customers/${c.id}`)}
                  >
                    <td className="px-4 py-3">
                      <div className="font-medium truncate max-w-[280px]">{c.name}</div>
                    </td>
                    <td className="px-4 py-3">
                      <div className="truncate max-w-[200px]">{c.contact_name || "—"}</div>
                    </td>
                    <td className="px-4 py-3">
                      <div className="truncate max-w-[240px]">{c.email || "—"}</div>
                    </td>
                    <td className="px-4 py-3">
                      <div className="truncate max-w-[160px]">{c.phone || "—"}</div>
                    </td>
                    <td className="px-4 py-3">
                      <div className="truncate max-w-[360px]">{addr || "—"}</div>
                    </td>
                    <td className="px-4 py-3" onClick={(e) => e.stopPropagation()}>
                      <div className="flex gap-2">
                        <Link href={`/customers/${c.id}`}>
                          <a className="text-blue-600 hover:underline">View</a>
                        </Link>
                        <button
                          className="text-blue-600 hover:underline"
                          onClick={() => navigate(`/customers/${c.id}`)}
                        >
                          Edit
                        </button>
                      </div>
                    </td>
                  </tr>
                );
              })}
            </tbody>
          </table>
        </div>
      )}

      <CustomerModal open={open} onOpenChange={setOpen} />
    </div>
  );
}
